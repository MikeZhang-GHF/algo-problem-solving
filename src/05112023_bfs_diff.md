```python
class Solution:
    def numIslands(self, grid: List[List[str]]) -> int:
        m, n = len(grid), len(grid[0])
        vis = set()

        def bfs(i, j):
            vis.add((i, j))
            q = [(i, j)]
            while q:
                tmp = q
                q = []
                for i, j in tmp:
                    # four directions and is land, expand
                    for x, y in (i+1, j), (i-1, j), (i, j+1), (i, j-1):
                        if 0 <= x < m and 0 <= y < n and grid[x][y] == '1' and (x,y) not in vis:
                            q.append((x,y))
                            vis.add((x,y))
                
        ans = 0
        for i in range(m):
            for j in range(n):
                if grid[i][j] == '1' and (i,j) not in vis:
                    bfs(i, j)
                    ans += 1
        return ans

```

### [Q2] -  LCP 79. 提取咒文

#### Solution:

- High Level:
  > **状态的思想**，(i,j,k)表示处在(i,j)位置，匹配到第k个字符。
  > 对于当前状态，可以匹配到第k+1个字符,转移到下一个状态(i,j,k+1)
  > 也可以不匹配，向四个方向拓展搜索空间,(i+1,j),(i-1,j),(i,j+1),(i,j-1)转移到下一个状态(x,y,k)

- Algorithm Knowledge:
  > BFS and understanding the state of BFS and state transition
  > Use BFS to find the shortest path

- Code Tricks:
  > representation of state
  > implementation of BFS shortest path

#### Code:

```python
class Solution:
    def extractMantra(self, matrix: List[str], mantra: str) -> int:
        m, n = len(matrix), len(matrix[0])
        # 开始的状态
        vis = set((0,0,0))
        q = [(0,0,0)]
        step = 1
        # BFS最小步数模型
        while q:
            tmp = q
            q = []
            for i, j, k in tmp:
                # 1. 当前字母匹配，可以提取
                if matrix[i][j] == mantra[k]:
                    if k == len(mantra) - 1:
                        return step
                    p = (i, j, k+1) 
                    q.append(p)
                    vis.add(p)
                else:
                    # 2. 不匹配，向四个方向拓展进行搜索匹配可能性
                    for x, y in (i+1,j), (i-1,j), (i,j+1), (i,j-1):
                        if 0 <= x < m and 0 <= y < n and (x,y,k) not in vis:
                            p = (x, y, k)
                            q.append(p)
                            vis.add(p)
            step += 1

        return -1
```

- [Q3] - 417. 太平洋大西洋水流问题
  >1 DFS 写法
  > - 使用st状态数组来标记是否访问过，1表示访问过太平洋，2表示访问过大西洋，3表示都访问过
```python
class Solution:
    def pacificAtlantic(self, heights: List[List[int]]) -> List[List[int]]:
        m, n = len(heights), len(heights[0])
        st = [[0] * n for _ in range(m)]

        def dfs(i, j, t):
            # 访问过
            if st[i][j] & t:
                return
            # 将这个状态标记进来，1=b001, 2=b010来标记两个源头
            st[i][j] |= t
            for x, y in (i+1,j), (i-1,j), (i,j+1), (i,j-1):
                if 0 <= x < m and 0 <= y < n and heights[x][y] >= heights[i][j]:
                    dfs(x, y, t)
        
        # 左边界起点
        for i in range(m):
            dfs(i, 0, 1)
        # 右边界起点
        for i in range(m):
            dfs(i, n - 1, 2)
        # 上边界起点
        for i in range(n):
            dfs(0, i, 1)
        # 下边界
        for i in range(n):
            dfs(m - 1, i, 2)

        ans = []
        for i in range(m):
            for j in range(n):
                if st[i][j] == 3:
                    ans.append([i, j])
        return ans
```

  ```python
class Solution:
    def pacificAtlantic(self, heights: List[List[int]]) -> List[List[int]]:
        m, n = len(heights), len(heights[0])
        def bfs(q, vis):
            # 使用一个队列的写法，Q2采用的是双数组的写法
            while q:
                i, j = q.popleft()
                for x, y in (i+1,j), (i-1,j), (i,j+1), (i,j-1):
                    if 0 <= x < m and 0 <= y < n and (x, y) not in vis and heights[x][y] >= heights[i][j]:
                        q.append((x, y))
                        vis.add((x, y))

        pacific = deque([(0, i) for i in range(n)] + [(i, 0) for i in range(1, m)])
        atlantic = deque([(i, n-1) for i in range(m-1)] + [(m-1, i) for i in range(0, n)])
        vis_p, vis_a = set(pacific), set(atlantic)
        bfs(pacific, vis_p)
        bfs(atlantic, vis_a)
        # 交集
        return list(vis_a & vis_p)
  ```
